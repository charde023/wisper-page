# 맥 로컬 전사 파이프라인 — 구현계획서 (Runbook)

> 상위 기획서: `docs/design/2026-07-02_맥-로컬-전사-파이프라인-기획서.md`
> 이 문서 = **바로 실행 가능한 구현 지시서**. 파일별 전체 코드 · 명령어 · Phase별 게이트 · 롤백 포함.
> 원칙: **산출물 계약 동일, 엔진만 교체.** 새로 만드는 건 파이썬 3개 + `config.json` 1개.

## 0. 결론 — 무엇을 만들고 어떤 순서로

| 신규 파일 | 역할 | 대체 대상 |
|---|---|---|
| `workflow/youtube/config.json` | 맥 경로·모델 오버라이드 | (신규) |
| `workflow/youtube/mac/transcribe_mlx.py` | ★ mlx-whisper 전사 코어 | `_template/transcribe.py`(CUDA) |
| `workflow/youtube/mac/yt_fetch.py` | yt-dlp → audio.wav + youtube.json | `yt_fetch.ps1` |
| `workflow/youtube/mac/run_youtube.py` | 큐 오케스트레이터(멱등·검증) | `run_youtube.ps1` + `transcribe.ps1` |

**빌드 순서(의존)**: `P0 환경` → ( `P1 엔진` ∥ `P2 페치` ) → `P3 오케스트레이터` → `P4 E2E` → `P5 고도화·스케줄`.
P1/P2는 독립이라 **병렬**(소넷 실행자 2개 동시 위임 가능).

## 2. Phase 1 — 전사 코어 `mac/transcribe_mlx.py`

기존 `_template/transcribe.py`와 **동일한 산출물**(`transcript.txt`·`segments.json`·`transcript.srt`·`transcribe.done`·`TRANSCRIBE_OK`)을 낸다. 성공 판정은 종료코드가 아니라 **산출물 검증**(기존 `.ps1` 철학 흡수).

```python
"""Transcribe audio with mlx-whisper (Apple MLX / Metal GPU).
_template/transcribe.py와 동일한 산출물 계약을 지킨다. 성공=산출물 검증."""
from __future__ import annotations
import argparse, json, sys, time
from pathlib import Path


def fmt_ts(s: float) -> str:
    h = int(s // 3600); m = int((s % 3600) // 60); x = s - h * 3600 - m * 60
    return f"{h:02d}:{m:02d}:{x:06.3f}".replace(".", ",")


def log(ws: Path, msg: str) -> None:
    line = f"[{time.strftime('%H:%M:%S')}] {msg}\n"
    sys.stdout.write(line); sys.stdout.flush()
    with (ws / "progress.log").open("a", encoding="utf-8") as f:
        f.write(line)


def main(argv=None) -> int:
    ap = argparse.ArgumentParser(description="mlx-whisper transcription")
    ap.add_argument("--workspace", type=Path, required=True)
    ap.add_argument("--language", default="en", help='언어코드 또는 "auto"')
    ap.add_argument("--model", default="mlx-community/whisper-large-v3-turbo")
    a = ap.parse_args(argv)

    ws = a.workspace.resolve()
    audio = ws / "audio.wav"
    if not audio.exists() or audio.stat().st_size <= 1024:
        print(f"ERROR: audio 없음/과소: {audio}", file=sys.stderr); return 1

    lang = None if a.language == "auto" else a.language
    # 이전 run 센티넬 제거(혼동 방지)
    (ws / "transcribe.done").unlink(missing_ok=True)
    log(ws, f"start model={a.model} lang={a.language} audio={audio.stat().st_size}B")

    import mlx_whisper  # 무거운 임포트는 여기(--help는 가볍게)
    t0 = time.time()
    r = mlx_whisper.transcribe(
        str(audio),
        path_or_hf_repo=a.model,
        language=lang,
        word_timestamps=True,
        condition_on_previous_text=False,  # 긴 강연 루프 방지(현행과 동일 철학)
    )
    segs = [
        {"id": i + 1, "start": s["start"], "end": s["end"], "text": s["text"].strip()}
        for i, s in enumerate(r.get("segments", []))
    ]
    if not segs:
        print("ERROR: 세그먼트 0개", file=sys.stderr); return 1
    duration = segs[-1]["end"]

    (ws / "transcript.txt").write_text(
        "\n".join(s["text"] for s in segs) + "\n", encoding="utf-8")
    (ws / "segments.json").write_text(json.dumps(
        {"model": a.model, "device": "mlx", "duration": duration, "segments": segs},
        ensure_ascii=False, indent=2), encoding="utf-8")
    with (ws / "transcript.srt").open("w", encoding="utf-8") as f:
        for s in segs:
            f.write(f"{s['id']}\n{fmt_ts(s['start'])} --> {fmt_ts(s['end'])}\n{s['text']}\n\n")
    (ws / "transcribe.done").write_text(
        f"segments={len(segs)} duration={duration:.1f}s model={a.model} device=mlx\n",
        encoding="utf-8")

    el = time.time() - t0
    rt = duration / el if el else 0
    log(ws, f"done {len(segs)} segs in {el:.1f}s ({rt:.1f}x realtime)")
    print(f"TRANSCRIBE_OK segments={len(segs)}")

    # manifest 스탬프(있으면)
    try:
        sys.path.insert(0, str(ws.parents[1] / "workflow" / "lib"))
        from manifest import set_stage  # type: ignore
        set_stage(ws, "transcribed")
    except Exception:
        pass
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

## 3. Phase 2 — 페치 `mac/yt_fetch.py`

`yt_fetch.ps1`을 1:1 이식. `yt_lib`의 헬퍼(`run_ytdlp`·`load_config`) 재사용.

```python
"""yt_fetch.ps1 이식: URL → workspaces/yt-<id>/audio.wav(16k mono) + youtube.json."""
from __future__ import annotations
import argparse, subprocess, sys
from pathlib import Path

YT_DIR = Path(__file__).resolve().parent.parent          # workflow/youtube
sys.path.insert(0, str(YT_DIR))
from yt_lib import load_config, run_ytdlp, ytdlp_path      # noqa: E402


def resolve_id(url: str, js: str) -> str | None:
    r = run_ytdlp(["--skip-download", "--no-warnings", "--print", "%(id)s", url], js)
    out = (r.stdout or "").strip().splitlines()
    return out[0].strip() if out else None


def main(argv=None) -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--url", required=True)
    ap.add_argument("--root", default=str(YT_DIR.parents[1]))  # 프로젝트 루트
    ap.add_argument("--force", action="store_true")
    a = ap.parse_args(argv)

    cfg = load_config(); js = cfg.get("ytJsRuntime", "node")
    vid = resolve_id(a.url, js)
    if not vid:
        print(f"ERROR: id 해석 실패 {a.url}", file=sys.stderr); return 1

    ws = Path(a.root) / "workspaces" / f"yt-{vid}"
    ws.mkdir(parents=True, exist_ok=True)
    (ws / ".source-url").write_text(a.url, encoding="utf-8")
    audio = ws / "audio.wav"

    if audio.exists() and audio.stat().st_size > 1024 and not a.force:
        print(f"audio.wav 존재 → 스킵 ({audio})")
    else:
        print("downloading audio (16kHz mono wav)...")
        subprocess.run([
            ytdlp_path(), "--js-runtimes", js, "-x", "--audio-format", "wav",
            "--postprocessor-args", "-ar 16000 -ac 1",
            "--write-info-json", "--no-warnings",
            "-o", str(ws / "audio.%(ext)s"), a.url,
        ], check=True)
        if not audio.exists() or audio.stat().st_size <= 1024:
            print("ERROR: audio.wav 미생성/과소 (throttle 시 --remote-components ejs:github)",
                  file=sys.stderr); return 1

    # youtube.json (메타 + 프로비넌스)
    prov = YT_DIR / "extract_provenance.py"
    if prov.exists():
        subprocess.run([sys.executable, str(prov), "--workspace", str(ws)], check=False)

    print(str(ws))   # 오케스트레이터가 마지막 줄로 캡처
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

## 4. Phase 3 — 오케스트레이터 `mac/run_youtube.py`

`run_youtube.ps1`+`transcribe.ps1` 이식: 큐 → (페치 → 전사) 직렬, **멱등**(`transcript.txt` 있으면 스킵), **산출물 검증**으로 성공 판정, 요약 출력.

```python
"""run_youtube.ps1 이식: 큐(JSON) 또는 URL 목록 → 페치 → 전사 배치."""
from __future__ import annotations
import argparse, json, subprocess, sys
from pathlib import Path

YT_DIR = Path(__file__).resolve().parent.parent
MAC = YT_DIR / "mac"
sys.path.insert(0, str(YT_DIR))
from yt_lib import load_config, run_ytdlp  # noqa: E402


def urls_from_queue(p: Path) -> list[str]:
    q = json.loads(p.read_text(encoding="utf-8"))
    out = []
    for v in q.get("videos", []):
        out.append(v.get("url") or f"https://youtu.be/{v['id']}")
    return out


def main(argv=None) -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--queue"); ap.add_argument("--url", nargs="*")
    ap.add_argument("--root", default=str(YT_DIR.parents[1]))
    ap.add_argument("--limit", type=int, default=0)
    ap.add_argument("--model", default=None)
    ap.add_argument("--language", default=None)
    a = ap.parse_args(argv)

    cfg = load_config()
    model = a.model or cfg.get("whisperModel", "mlx-community/whisper-large-v3-turbo")
    lang = a.language or cfg.get("whisperLanguage", "en")
    js = cfg.get("ytJsRuntime", "node")

    urls = urls_from_queue(Path(a.queue)) if a.queue else (a.url or [])
    if not urls:
        print("ERROR: --queue 또는 --url 필요", file=sys.stderr); return 1
    if a.limit:
        urls = urls[: a.limit]
    print(f"queued {len(urls)} video(s)  model={model} lang={lang}")

    results = []
    for i, u in enumerate(urls, 1):
        print(f"\n{'='*56}\n[{i}/{len(urls)}] {u}\n{'='*56}")
        r = run_ytdlp(["--skip-download", "--no-warnings", "--print", "%(id)s", u], js)
        vid = (r.stdout or "").strip().splitlines()
        vid = vid[0].strip() if vid else None
        if not vid:
            results.append(("FAILED", u, "id 해석 실패")); continue
        ws = Path(a.root) / "workspaces" / f"yt-{vid}"

        # 멱등: 이미 전사됨?
        tr = ws / "transcript.txt"
        if tr.exists() and tr.stat().st_size > 0:
            print("[transcribe] transcript.txt 존재 → 캐시 스킵")
            results.append(("cached", str(ws), "-")); continue

        # 페치
        print("[fetch] ...")
        try:
            subprocess.run([sys.executable, str(MAC / "yt_fetch.py"),
                            "--url", u, "--root", a.root], check=True)
        except subprocess.CalledProcessError as e:
            results.append(("FAILED", str(ws), f"fetch: {e}")); continue
        if not (ws / "audio.wav").exists():
            results.append(("FAILED", str(ws), "no audio.wav")); continue

        # 전사
        print(f"[transcribe] {model} / mlx ...")
        subprocess.run([sys.executable, str(MAC / "transcribe_mlx.py"),
                        "--workspace", str(ws), "--language", lang,
                        "--model", model], check=False)
        # 산출물 검증 = 진짜 성공 신호
        ok = (ws / "transcript.txt").exists() and (ws / "transcript.txt").stat().st_size > 0 \
            and (ws / "transcribe.done").exists()
        results.append(("transcribed" if ok else "FAILED", str(ws),
                        "-" if ok else "no transcript"))

    n_ok = sum(1 for s, *_ in results if s in ("transcribed", "cached"))
    print(f"\n{'='*56}\nSUMMARY  {n_ok}/{len(results)} ok\n{'='*56}")
    for st, ws, note in results:
        print(f"  {st:<12} {ws}" + (f"  ({note})" if note != '-' else ""))
    print("\nnext: transcript.txt+youtube.json → 학습노트 작성 → rebuild_index.py")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

## 6. Phase 5 — 고도화·스케줄 (선택, 우선순위순)

### 6.1 자막 우선 탐침 (전사 스킵 가능)
`yt_fetch.py`에 프리체크 추가: `yt-dlp --list-subs`로 ko/en 트랙 있으면 `--write-subs --sub-lang ko --skip-download`로 `.vtt` 받아 전사 대체(무료·정확). 번인이면 트랙 없음 → Whisper 진행.

### 6.2 야간 스케줄 (launchd)
`~/Library/LaunchAgents/kr.techbridge.nightly.plist` — 매일 밤 `rss_watch → run_youtube(신규 큐) → 알림`. 앱 없이 도는 순수 CLI라 launchd 적합(시그마 윈도우 스케줄 은퇴).


## 8. 롤백 / 리스크

| 리스크 | 대응 |
|---|---|
| yt-dlp가 YouTube nsig로 실패 | `brew upgrade yt-dlp`; `--remote-components ejs:github` 폴백; node/deno 런타임 확인 |
| turbo 정확도 부족(고유명사) | 모델 ID만 `whisper-large-v3`로 교체(코드 무변경) |
| mlx-whisper 세그먼트 없음/무음 | silero-VAD 전처리 추가(D6) |
| 신규 스크립트 회귀 | `.ps1`·`_template/transcribe.py`는 **삭제 안 함**(시그마 폴백 보존). 맥은 `mac/` 하위로 격리 |
| 볼트 경로 오타 | `config.json`의 `vaultNoteDir` 존재 검증(`rebuild_index.py` 첫 줄) |

**롤백**: `mac/` 폴더 + `config.json`만 제거하면 기존 윈도우 파이프라인 원상 복구(다운스트림 무영향).
